# Where local *cot feeders (adsbcot, aiscot, …) send Cursor on Target messages.
# Default: Charontak ingress on localhost. Upstream mesh / TAK Server forwarding is
# configured in /etc/charontak.ini (Cockpit → Charontak at https://<host>/admin/).
# See: https://pytak.readthedocs.io/en/stable/configuration/
COT_URL=udp+wo://127.0.0.1:28087


# For Mesh SA & multicast COT_URLs: Which network interface (wifi, ethernet) do we bind to?
# Options are IP, or 0.0.0.0 for all. Default is WiFi AP IP: 10.41.0.1
PYTAK_MULTICAST_LOCAL_ADDR=10.41.0.1

# DEPRECATED: Legacy setting.
WIFI_AP_IP=10.41.0.1

# 1090 MHz ADS-B decoder: readsb or dump1090_fa (only one may be enabled). Must match image build
# (vars.yml aryaos_adsb_decoder). Changing this alone does not reconfigure systemd — see docs/config.md.
ARYAOS_ADSB_DECODER=readsb

# 1090 MHz ADS-B JSON directory (readsb --write-json and dump1090-fa --write-json). adsbcot FEED_URL uses aircraft.json here.
ARYAOS_ADSB_JSON_DIR=/run/adsb

# Network-facing CoT services to restart if network state changes.
# Keep local decoders, GPS, UI, and Bluetooth bridge out of this list; restarting
# them during boot interrupts radio ingest and Bluetooth pairing/RFCOMM.
AOS_SERVICES="charontak gpstak aiscot lincot adsbcot dronecot adsbxcot aprscot spotcot"

# RTL-SDR USB serial string for dump978-fa (UAT / 978 MHz). Factory default matches Nooelec
# NESDR Nano 3 “978” EEPROM preset: stx:978:0 — do not use the same serial as the 1090 path
# (readsb / dump1090-fa use adsb_sdr_sn in Ansible / image build). Override here on a running
# system, then: sudo systemctl restart dump978-fa
ARYAOS_UAT_RTL_SERIAL=stx:978:0


# Device identity — set on first boot by aryaos-firstboot.sh; do not modify. NO STEP.
# DEVICE_SUFFIX: last 4 hex of machine-id (or MAC); hostname aryaos-xxxx and WiFi AryaOS-xxxx.
DEVICE_SUFFIX=""

# COT_HOST_ID: functional source id stamped into the CoT _flow-tags_ and remarks by
# the PyTAK *cot tools. Set on first boot to aryaos-<suffix>; override for a custom name.
COT_HOST_ID=""
